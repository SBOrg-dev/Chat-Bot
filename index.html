async function send() { let inputField = document.getElementById("msg"); let text = inputField.value; let rispostaDiv = document.getElementById("risposta"); if (!text) return; // Non inviare se il campo è vuoto rispostaDiv.innerText = "Sto pensando..."; inputField.value = ""; // Pulisce l'input dopo l'invio try { const response = await fetch("/api/chat", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: text }) }); if (!response.ok) throw new Error("Errore di rete"); const data = await response.json(); // Verifica che i dati esistano prima di scriverli if (data.choices && data.choices[0]) { rispostaDiv.innerText = data.choices[0].message.content; } else { rispostaDiv.innerText = "Risposta non valida dal server."; } } catch (error) { rispostaDiv.innerText = "Si è verificato un errore: " + error.message; } }
