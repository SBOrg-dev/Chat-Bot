async function send() { const inputField = document.getElementById("msg"); const rispostaDiv = document.getElementById("risposta"); const text = inputField.value.trim(); if (!text) return; rispostaDiv.innerText = "Sto pensando..."; inputField.value = ""; try { const response = await fetch("/api/chat", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message: text }) }); if (!response.ok) { throw new Error("Errore server"); } const data = await response.json(); if (data.choices && data.choices.length > 0) { rispostaDiv.innerText = data.choices[0].message.content; } else { rispostaDiv.innerText = "Risposta non valida dal server."; } } catch (error) { rispostaDiv.innerText = "Errore: " + error.message; } }
